Python Code

from anna_protocol.client import ANNAClient, Reasoning, ReasoningStep, Metadata
import os
from dotenv import load_dotenv

load_dotenv()
client = ANNAClient(private_key=os.getenv('PRIVATE_KEY'), network='polygon-amoy')

# Funs.ai AI reasoning
reasoning = Reasoning(
    input="KYC for Alice Silva: ID + selfie (97% match)",
    reasoning_steps=[
        ReasoningStep(1, "Biometric Analysis", "Selfie match: 97% with ID."),
        ReasoningStep(2, "Document Check", "Valid ID (SP 2023)."),
        ReasoningStep(3, "SocialFi Risk", "No OFAC/PEP hits.")
    ],
    conclusion="Approve KYC. Grant 'Verified Creator' badge.",
    confidence=0.95
)

metadata = Metadata(
    external_id="FUNS-KYC-2025-001",
    document_type="socialfi_creator_onboarding",
    client_name="Alice Silva",
    custom_fields={"badge": "Verified Creator", "risk": 0.05}
)

result = client.create_attestation(
    content="KYC approved: Alice Silva",
    reasoning=reasoning,
    metadata=metadata,
    category="socialfi_kyc"
)

print(f"ID: {result.attestation_id} | Tx: {result.tx_hash}")
